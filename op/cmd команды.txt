cd и название папки -= туда перезод

dir -= узнать что в папке

ls - тоде самое

mkdir название папки -= создает папку

cd .. -= назад в папку

rm -= -r удаление директории 
rm -= удаление файла

cp -= название файла =- копирует файл

пример: cp txt.txt ../названеи папки
cp -r копирует дирректорию

mv -= перместить файл
пример: mr txt.txt название папки

touch txt.txt -= создает файл

head имя файла -= показывает начало файлаече
tail -= показывает когец файла

echo -= и что-то просото написать оно и выведет

pwd -= показывает полный путь 

cat -= показывает то что в файле
